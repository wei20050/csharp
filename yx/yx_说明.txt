使用方法: 
1.引用yx.dll
2.程序开头  using yx;
3.注册钩子
private void Form1_Load(object sender, EventArgs e)
{
    KeyHook key = new KeyHook();
    key.SetHook();
    key.OnKeyPressEvent += (o, args) =>
    {
        MessageBox.Show(args.KeyChar.ToString());
    };
}
